import {
  Meta,
  Story,
  Canvas,
  ArgsTable,
  ColorPalette,
  ColorItem
} from '@storybook/addon-docs/blocks';
import { IconGallery } from 'storybook-icon-gallery';
import IconSvgCloud from 'asset/images/svg/icon-cloud.svg';
import IconSvgForm from 'asset/images/svg/icon-form.svg';
import IconSvgHome from 'asset/images/svg/icon-home.svg';
import IconSvgMail from 'asset/images/svg/icon-mail.svg';
import IconSvgMenu from 'asset/images/svg/icon-menu.svg';
import IconSvgMessage from 'asset/images/svg/icon-message.svg';
import IconSvgOffice from 'asset/images/svg/icon-office.svg';
import IconPngLogo from 'asset/images/png/logo.png';

export const IconArgsType = {
  IconSvgElement: {
    table: { disable: true }
  },
  color: { control: { type: 'color' } }
};

export function renderItem(
  name,
  Icon1,
  Size1 = 25,
  Color1 = '#000',
  Icon2,
  Size2 = 25,
  Color2 = '#000'
) {
  return (
    <IconGallery.Variants name={name}>
      <IconGallery.Variant size={Size1}>
        {Icon1 && <Icon1 color={Color1} />}
      </IconGallery.Variant>
      <IconGallery.Variant size={Size2}>
        {Icon2 && <Icon2 color={Color2} />}
      </IconGallery.Variant>
    </IconGallery.Variants>
  );
}

<Meta title='Icons/svg/plugin' component={Icon} argTypes={IconArgsType} />
